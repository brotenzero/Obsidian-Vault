--1 - 5 #축구쿼리문제 

-- 001. 전체 축구팀 목록을 팀이름 오름차순으로 출력하시오
select * from team order by team_name collate "C" asc;

-- 002. 플레이어의 포지션 종류를 나열하시오. 단 중복은 제거하고, 포지션이 없으면 빈공간으로 두시오
select position from player where position is null union all select distinct position from player where position is not null;
-- 003. 플레이어의 포지션 종류를 나열하시오. 단 중복은 제거하고, 포지션이 없으면 '신입' 으로 기재하시오
select distinct coalesce(position, '신입') from player;

-- 004. 수원팀에서 골키퍼(GK)의 이름을 모두 출력하시오. 단 수원팀 ID는 K02 입니다.

select player_name
from player p join team t
on p.team_id = t.team_id
where t.region_neme = '수원'  and p.positionn = 'GK';

select player_name
from player p join team t
on p.team_id = t.team_id
where t.region_neme = '수원'  and p.positionn = 'GK';

-- 005. 수원팀에서 성이 고씨이고 키가 170 이상인 선수를 출력하시오. 단 수원팀 ID는 K02 입니다.
select * from player where team_id = 'K02' and left(player_name, 1) = '고' and height >= 170;


-- 005-1. 수원팀의 ID 는 ?
select team_id from team where region_name = '수원';

-- 005-2. 수원팀에서 성이 고씨이고 키가 170 이상인 선수를 출력하시오. (서브쿼리)
SELECT player_name, height FROM player 
WHERE team_id IN (SELECT team_id FROM team WHERE region_name = '수원') AND LEFT(player_name, 1) = '고' AND height >= 170;
-- 문제 6
-- 다음 조건을 만족하는 선수명단을 출력하시오
-- 소속팀이 삼성블루윙즈이거나
-- 드래곤즈에 소속된 선수들이어야 하고,
-- 포지션이 미드필더(MF:Midfielder)이어야 한다.
-- 키는 170 센티미터 이상이고 180 이하여야 한다.
select player_name, team_id, height, position from player where (team_id ='K07' or team_id = 'K02') and position = 'MF' and height >= 170 and height <= 180;
